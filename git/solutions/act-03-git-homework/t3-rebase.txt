$ echo "Initial content" > file.txt
$ git add file.txt
$ git commit -m "Initial commit"
[main (root-commit) 1a2b3c4] Initial commit
 1 file changed, 1 insertion(+)
 create mode 100644 file.txt

# Create a topic branch 'feature'
$ git branch feature
$ git checkout feature
Switched to branch 'feature'

# Make commits on feature branch
$ echo "Feature commit 1" >> file.txt
$ git add file.txt
$ git commit -m "Feature commit 1"
[feature 2b3c4d5] Feature commit 1
 1 file changed, 1 insertion(+)

$ echo "Feature commit 2" >> file.txt
$ git add file.txt
$ git commit -m "Feature commit 2"
[feature 3c4d5e6] Feature commit 2
 1 file changed, 1 insertion(+)

# Switch back to main and make a new commit
$ git checkout main
Switched to branch 'main'

$ echo "Main branch update" >> file.txt
$ git add file.txt
$ git commit -m "Update on main branch"
[main 4d5e6f7] Update on main branch
 1 file changed, 1 insertion(+)

# Show git log before rebase
$ git log --oneline --decorate --graph --all
* 4d5e6f7 (HEAD -> main) Update on main branch
| * 3c4d5e6 (feature) Feature commit 2
| * 2b3c4d5 Feature commit 1
|/
* 1a2b3c4 Initial commit

# Rebase feature onto main
$ git checkout feature
Switched to branch 'feature'

$ git rebase main
First, rewinding head to replay your work on top of it...
Applying: Feature commit 1
Applying: Feature commit 2

# Show git log after rebase
$ git log --oneline --decorate --graph --all
* 3c4d5e6 (HEAD -> feature) Feature commit 2
* 2b3c4d5 Feature commit 1
* 4d5e6f7 (main) Update on main branch
* 1a2b3c4 Initial commit
