1. Full combined workflow file content (.github/workflows/super-linter-docker-build-and-push.yml)

name: Lint Code Base and Docker Build and Push

on:
  push:
    branches: ["main"]
  pull_request:
    branches: ["main"]
jobs:
  run-lint:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout code
        uses: actions/checkout@v3
        with:
          # Full git history is needed to get a proper list of changed files within `super-linter`
          fetch-depth: 0

      - name: Lint Code Base
        uses: github/super-linter@v4
        env:
          VALIDATE_ALL_CODEBASE: false
          DEFAULT_BRANCH: "main"
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
          DISABLE_RULES: "space-tab-mixed"
  docker:
    runs-on: ubuntu-latest
    needs: run-lint
    steps:
      - name: Set up QEMU
        uses: docker/setup-qemu-action@v3
      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3
      - name: Verify Docker CLI
        run: docker --version
      - name: Login to Docker Hub
        uses: docker/login-action@v3
        with:
          username: ${{ secrets.DOCKERHUB_USERNAME }}
          password: ${{ secrets.DOCKERHUB_TOKEN }}
      - name: Build and push
        uses: docker/build-push-action@v5
        with:
          push: true
          tags: aristikj/kii-demo-01:latest

2. Dependency requirement

The docker job is configured with:
needs: [super-lint]

This ensures Build and Push Docker Image runs only after Super Lint succeeds.

3. Names of configured repository secrets

- DOCKERHUB_USERNAME
- DOCKERHUB_TOKEN
